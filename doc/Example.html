def update(state)
  case state.current_state
    when :PORT_COMPLETE
      puts "#{state.transport} port #{state.port} #{state.description(state.port_result_code)}"
    when :PORT_START
    when :PORT_TICK
    when :SCAN_START
    when :START_FAILURE
    when :SCAN_COMPLETE
    else
  end
end
